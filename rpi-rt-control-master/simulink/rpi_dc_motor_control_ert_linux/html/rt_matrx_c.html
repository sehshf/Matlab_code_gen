  198   /* Function: mxCreateCharMatrixFromStrings ====================================
  199    * Abstract:
  200    *	Create a string array initialized to the strings in str.
  201    */
  202   
  203   #ifdef __cplusplus
  204   extern "C" {
  205   #endif
  206   
  207   mxArray *rt_mxCreateCharMatrixFromStrings(int_T m, const char_T **str)
  208   {
  209       int_T nchars;
  210       int_T i, n;
  211       mxArray *pa;
  212   
  213       n = 0;
  214       for (i = 0; i < m; ++i) {
  215   	nchars = (int_T)strlen(str[i]);
  216   	if (nchars > n) {
  217   	    n = nchars;
  218   	}
  219       }
  220       /*LINTED E_PASS_INT_TO_SMALL_INT*/
  221       pa = (mxArray *)malloc((m*n+2)*sizeof(real_T));
  222       if(pa!=NULL) {
  223   	mxChar *chars;
  224   	int_T  j;
  225   	_mxSetM(pa, m);
  226   	_mxSetN(pa, n);
  227   	chars = mxGetPr(pa);
  228   	for (j = 0; j < m; ++j) {
  229   	    const char_T *src  = str[j];
  230   	    mxChar *dest = chars + j;
  231   
  232   	    nchars = (int_T)strlen(src);
  233   	    i = nchars;
  234   	    while (i--) {
  235   		*dest = *src++;
  236   		 dest += m;
  237   	    }
  238   	    i = n - nchars;
  239   	    while (i--) {
  240   		*dest = 0.0;
  241   		dest += m;
  242   	    }
  243   	}
  244       }
  245       return pa;
  246   } /* end mxCreateCharMatrixFromStrings */
  247   
  248   #ifdef __cplusplus
  249   }

  252   /* Function: mxCreateString ===================================================
  253    * Abstract:
  254    *	Create a 1-by-n string array initialized to null terminated string
  255    *	where n is the length of the string.
  256    */
  257   #ifdef __cplusplus
  258   extern "C" {
  259   #endif
  260   
  261   mxArray *rt_mxCreateString(const char *str)
  262   {
  263       int_T   len = (int_T)strlen(str);
  264       /*LINTED E_PASS_INT_TO_SMALL_INT*/
  265       mxArray *pa = (mxArray *)malloc((len+2)*sizeof(real_T));
  266   
  267       if(pa!=NULL) {
  268   	real_T *pr;
  269   	const unsigned char *ustr_ptr = (const unsigned char *) str;
  270   
  271   	_mxSetM(pa, 1);
  272   	_mxSetN(pa, len);
  273   	pr = mxGetPr(pa);
  274   	while (len--) {
  275               *pr++ = (real_T)*ustr_ptr++;
  276           }
  277       }
  278       return(pa);
  279   
  280   } /* end mxCreateString */
  281   
  282   #ifdef __cplusplus
  283   }

  286   /* Function: mxCreateDoubleMatrix =============================================
  287    * Abstract:
  288    *	Create a two-dimensional array to hold real_T data,
  289    *	initialize each data element to 0.
  290    */
  291   /*LINTLIBRARY*/
  292   
  293   #ifdef __cplusplus
  294   extern "C" {
  295   #endif
  296   
  297   mxArray *rt_mxCreateDoubleMatrix(int m, int n, mxComplexity flag)
  298   {
  299       if (flag == mxREAL) {
  300           mxArray *pa = (mxArray *)calloc(m*n+2, sizeof(real_T));
  301           if(pa!=NULL) {
  302               _mxSetM(pa, m);
  303               _mxSetN(pa, n);
  304           }
  305           return(pa);
  306       } else {
  307           return(NULL);
  308       }
  309   
  310   } /* end mxCreateDoubleMatrix */
  311   
  312   #ifdef __cplusplus
  313   }

  340   /* Function: mxDuplicateArray =================================================
  341    * Abstract:
  342    *	Make a deep copy of an array, return a pointer to the copy.
  343    */
  344   #ifdef __cplusplus
  345   extern "C" {
  346   #endif
  347   
  348   
  349   mxArray *rt_mxDuplicateArray(const mxArray *pa)
  350   {
  351       /*LINTED E_ASSIGN_INT_TO_SMALL_INT*/
  352       size_t   nbytes = (mxGetNumberOfElements(pa)+2)*mxGetElementSize(pa);
  353       mxArray *pcopy = (mxArray *)malloc(nbytes);
  354   
  355       if (pcopy!=NULL) {
  356   	(void)memcpy(pcopy, pa, nbytes);
  357       }
  358       return(pcopy);
  359   
  360   } /* end mxDuplicateArray */
  361

  363   #ifdef __cplusplus
  364   }

  418   /* Function: mxGetString ======================================================
  419    * Abstract:
  420    *	Converts a string array to a C-style string.
  421    */
  422   
  423   
  424   #ifdef __cplusplus
  425   extern "C" {
  426   #endif
  427   
  428   
  429   int_T rt_mxGetString(const mxArray *pa, char_T *buf, int_T buflen)
  430   {
  431       int_T        nchars;
  432       const real_T *pr;
  433       char_T       *pc;
  434       int_T        truncate = 0;
  435   
  436       nchars = (int_T)mxGetNumberOfElements(pa);
  437       if (nchars >= buflen) {
  438   	/* leave room for null byte */
  439   	nchars = buflen - 1;
  440   	truncate = 1;
  441       }
  442       pc = buf;
  443       pr = mxGetPr(pa);
  444       while (nchars--) {
  445   	*pc++ = (char) (*pr++ + .5);
  446       }
  447       *pc = '\0';
  448       return truncate;
  449   } /* end mxGetString */
